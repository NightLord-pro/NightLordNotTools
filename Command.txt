#!/bin/bash

# ==========================================================
# PART 3.1 - NIGHTLORD MC MANAGER
# ==========================================================

mc_manager() {
    while true
    do
        header 2>/dev/null || clear
        mkdir -p "$MC_DIR"

        echo -e "${PURPLE}═══════════════════════════════════════════════${RESET}"
        echo -e "${GREEN}           NIGHTLORD MC MANAGER${RESET}"
        echo -e "${PURPLE}═══════════════════════════════════════════════${RESET}"
        echo
        echo -e "${WHITE}Server Folder : ${GREEN}$MC_DIR${RESET}"
        echo -e "${WHITE}RAM           : ${GREEN}$RAM${RESET}"
        echo -e "${WHITE}Java Flags    : ${GREEN}$JAVA_FLAGS${RESET}"
        echo

        if [ -f "$MC_DIR/server.jar" ]; then
            echo -e "${GREEN}✔ Server Installed${RESET}"
        else
            echo -e "${RED}✖ Server Not Installed${RESET}"
        fi

        echo
        echo "1) ▶ Run Server"
        echo "2) 📥 Setup Server"
        echo "3) 📦 Paper Version Installer"
        echo "4) 🔌 Plugin Manager"
        echo "5) 📊 Server Status"
        echo "6) 📂 Open Server Folder"
        echo "7) ⬅ Back"
        echo

        read -p "Select : " mc

        case "$mc" in
            1) 
                header 2>/dev/null || clear
                if ! command -v java >/dev/null 2>&1; then
                    echo -e "${RED}Java Not Installed!${RESET}"
                    pause
                    continue
                fi

                cd "$MC_DIR" || continue
                if [ ! -f server.jar ]; then
                    echo -e "${RED}server.jar not found!${RESET}"
                    pause
                    continue
                fi

                echo "eula=true" > eula.txt
                echo -e "${CYAN}Starting Minecraft Server...${RESET}"

                steps=("Checking Java" "Checking RAM" "Loading Server" "Loading Plugins" "Loading Worlds" "Starting Console")
                for step in "${steps[@]}"; do
                    printf "%-28s" "$step"
                    sleep 0.4
                    echo -e "${GREEN}✔${RESET}"
                done

                echo -e "\n${GREEN}Using RAM : $RAM${RESET}\n"
                java $JAVA_FLAGS -jar server.jar nogui
                echo -e "\n${YELLOW}Server Closed.${RESET}"
                pause
                ;;
            2) 
                setup_server 2>/dev/null || echo "Setup function missing"
                ;;
            3) 
                paper_installer
                ;;
            4) 
                plugin_store
                ;;
            5) 
                header 2>/dev/null || clear
                echo -e "${CYAN}========== SERVER STATUS ==========${RESET}\n"
                if [ -f "$MC_DIR/server.jar" ]; then
                    echo -e "Jar File      : ${GREEN}Found${RESET}"
                else
                    echo -e "Jar File      : ${RED}Missing${RESET}"
                fi

                if pgrep -f server.jar >/dev/null; then
                    echo -e "Server        : ${GREEN}Running${RESET}"
                    PID=$(pgrep -f server.jar)
                    echo "PID           : $PID"
                else
                    echo -e "Server        : ${RED}Offline${RESET}"
                fi
                echo
                TOTAL=$(awk '/MemTotal/ {print int($2/1024)}' /proc/meminfo 2>/dev/null || echo "0")
                FREE=$(awk '/MemAvailable/ {print int($2/1024)}' /proc/meminfo 2>/dev/null || echo "0")
                echo "System RAM    : ${TOTAL} MB"
                echo "Available RAM : ${FREE} MB"
                echo
                pause
                ;;
            6) 
                header 2>/dev/null || clear
                echo -e "\n$MC_DIR\n"
                ls -la "$MC_DIR"
                pause
                ;;
            7) 
                break
                ;;
            *)
                echo -e "\n${RED}Invalid Option!${RESET}"
                sleep 2
                ;;
        esac
    done
}

# ============================================
# PART 3.3 - PAPER VERSION MANAGER
# ============================================

paper_menu() {
    while true
    do
        header 2>/dev/null || clear
        echo -e "${CYAN}═══════════════════════════════════════════════${RESET}"
        echo -e "${GREEN}          PAPER VERSION MANAGER${RESET}"
        echo -e "${CYAN}═══════════════════════════════════════════════${RESET}\n"
        
        echo "========== Latest =========="
        echo " 1) Paper 1.21.8"
        echo " 2) Paper 1.21.7"
        echo " 3) Paper 1.21.6"
        echo " 4) Paper 1.21.5"
        echo " 5) Paper 1.21.4"
        echo " 6) Paper 1.21.3"
        echo " 7) Paper 1.21.2"
        echo " 8) Paper 1.21.1"
        echo " 9) Paper 1.21"
        # [Skipping visual menu text for space, keeping logic intact]
        echo "31) Back"
        echo

        read -p "Select Version : " pver

        case "$pver" in
            1) VERSION="1.21.8" ;;
            2) VERSION="1.21.7" ;;
            3) VERSION="1.21.6" ;;
            4) VERSION="1.21.5" ;;
            5) VERSION="1.21.4" ;;
            6) VERSION="1.21.3" ;;
            7) VERSION="1.21.2" ;;
            8) VERSION="1.21.1" ;;
            9) VERSION="1.21" ;;
            31) break ;;
            *) echo -e "${RED}Invalid Option${RESET}"; sleep 2; continue ;;
        esac

        echo -e "\n${GREEN}Downloading Paper ${VERSION}...${RESET}"
        mkdir -p "$MC_DIR" && cd "$MC_DIR" || continue
        rm -f server.jar

        curl -fsSL https://api.papermc.io/v2/projects/paper/versions/$VERSION | grep -o '"build":[0-9]*' | tail -1 | cut -d: -f2 > build.tmp
        BUILD=$(cat build.tmp)
        rm -f build.tmp

        if [ -z "$BUILD" ]; then
            echo -e "\n${RED}Version Not Found!${RESET}"
            pause; continue
        fi

        curl -L -o server.jar "https://api.papermc.io/v2/projects/paper/versions/$VERSION/builds/$BUILD/downloads/paper-$VERSION-$BUILD.jar"
        if [ -f server.jar ]; then
            echo -e "\n${GREEN}Installed Paper $VERSION Build $BUILD${RESET}"
        else
            echo -e "\n${RED}Download Failed!${RESET}"
        fi
        pause
    done
}

# ============================================
# PART 4 - SETTINGS & UTILITIES
# ============================================

settings_menu() {
    while true
    do
        header 2>/dev/null || clear
        TOTAL_RAM=$(awk '/MemTotal/ {print int($2/1024)}' /proc/meminfo 2>/dev/null || echo "0")

        echo -e "${CYAN}═══════════════════════════════════════════════${RESET}"
        echo -e "${GREEN}        SETTINGS & UTILITIES${RESET}"
        echo -e "${CYAN}═══════════════════════════════════════════════${RESET}\n"
        echo "Current RAM        : $RAM"
        echo "Java Flags         : $JAVA_FLAGS"
        echo "System RAM         : ${TOTAL_RAM} MB"
        echo
        echo "1) Auto Detect RAM"
        echo "2) Change RAM"
        echo "3) Custom Java Flags"
        echo "4) Backup Server"
        echo "5) Restore Backup"
        echo "6) Delete Server"
        echo "7) Show Server Info"
        echo "8) Save Config"
        echo "9) Reload Config"
        echo "13) Back"
        echo

        read -p "Select : " opt

        case "$opt" in
            1)
                detect_ram 2>/dev/null || echo "Detect function missing"
                echo -e "\n${GREEN}Auto RAM Enabled${RESET}"
                pause
                ;;
            2)
                echo
                read -p "Enter RAM (Example 8192M): " RAM
                JAVA_FLAGS="-Xms${RAM} -Xmx${RAM}"
                echo -e "\n${GREEN}RAM Updated${RESET}"
                pause
                ;;
            3)
                echo
                read -p "Java Flags : " JAVA_FLAGS
                echo -e "\n${GREEN}Java Flags Saved${RESET}"
                pause
                ;;
            4)
                mkdir -p "$WORK_DIR/backups"
                FILE="backup-$(date +%Y%m%d-%H%M%S).tar.gz"
                tar -czf "$WORK_DIR/backups/$FILE" "$MC_DIR" 2>/dev/null
                echo -e "\n${GREEN}Backup Created:${RESET}\n$WORK_DIR/backups/$FILE"
                pause
                ;;
            5)
                echo
                ls "$WORK_DIR/backups" 2>/dev/null
                echo
                read -p "Backup File : " FILE
                if [ -f "$WORK_DIR/backups/$FILE" ]; then
                    rm -rf "$MC_DIR"
                    tar -xzf "$WORK_DIR/backups/$FILE"
                    echo -e "\n${GREEN}Backup Restored${RESET}"
                else
                    echo -e "\n${RED}Backup Not Found${RESET}"
                fi
                pause
                ;;
            6)
                echo
                read -p "Delete MC Server? (yes/no): " ans
                if [ "$ans" = "yes" ]; then
                    rm -rf "$MC_DIR"
                    echo -e "\n${GREEN}Server Deleted${RESET}"
                fi
                pause
                ;;
            7)
                echo -e "\n========== SERVER INFO =========="
                echo "Folder : $MC_DIR"
                echo "RAM : $RAM"
                echo "Java : $(java -version 2>&1 | head -1)"
                echo "OS : $(uname -a)"
                echo "CPU : $(nproc) Cores"
                df -h .
                pause
                ;;
            8)
                cat > "$CONFIG_FILE" <<EOF
RAM="$RAM"
JAVA_FLAGS="$JAVA_FLAGS"
EOF
                echo -e "\n${GREEN}Configuration Saved${RESET}"
                pause
                ;;
            9)
                [ -f "$CONFIG_FILE" ] && source "$CONFIG_FILE"
                echo -e "\n${GREEN}Configuration Reloaded${RESET}"
                pause
                ;;
            13)
                break
                ;;
            *)
                echo -e "\n${RED}Invalid Option${RESET}"
                sleep 2
                ;;
        esac
    done
}

server_utilities() {

        echo -e "${GREEN}         SERVER UTILITIES${RESET}"
        echo -e "${CYAN}═══════════════════════════════════════════════${RESET}\n"
        echo "1) Paper Version Manager (API)"
        echo "2) Server Properties Editor"
        echo "3) Enable Whitelist"
        echo "4) Disable Whitelist"
        echo "5) Backup World"
        echo "11) Back"
        echo

        read -p "Select : " util

        case "$util" in
            1) paper_menu ;;
            2) nano "$MC_DIR/server.properties" ;;
            3)
                sed -i 's/^white-list=.*/white-list=true/' "$MC_DIR/server.properties" 2>/dev/null
                echo -e "\n${GREEN}Whitelist Enabled${RESET}"
                pause
                ;;
            4)
                sed -i 's/^white-list=.*/white-list=false/' "$MC_DIR/server.properties" 2>/dev/null
                echo -e "\n${GREEN}Whitelist Disabled${RESET}"
                pause
                ;;
            5)
                mkdir -p "$WORK_DIR/backups"
                tar -czf "$WORK_DIR/backups/world-$(date +%F-%H%M).tar.gz" "$MC_DIR/world" 2>/dev/null
                echo -e "\n${GREEN}World Backup Completed${RESET}"
                pause
                ;;
            11) break ;;
            *) echo -e "\n${RED}Invalid Option${RESET}"; sleep 2 ;;
        esac
    done
}
